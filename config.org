#+TITLE: Config
* Introduction
This configuration is built from scratch with a focus on performance and
compatibility. It is structured as a literate document so that each section
can be understood and modified independently. The configuration targets
Emacs 29+ on NixOS.

* Initialization
This section sets up the package manager and use-package. It must run first
since everything else depends on it.

#+BEGIN_SRC emacs-lisp
(require 'package)

(setq package-archives
      '(("melpa"  . "https://melpa.org/packages/")
        ("elpa"   . "https://elpa.gnu.org/packages/")
        ("nongnu" . "https://elpa.nongnu.org/packages/")))

(package-initialize)

(unless package-archive-contents
  (package-refresh-contents))

(require 'use-package)
(setq use-package-always-ensure t)

#+END_SRC

** Starter Screen
#+BEGIN_SRC emacs-lisp
;; Start page
(let ((buf (get-buffer-create "*start*")))
  (with-current-buffer buf
    (let ((inhibit-read-only t))
      (erase-buffer)
      (insert "\n\n")
      (insert (propertize
               "                     d8,
                    `8P

.d88b,.88P?88   d8P  88b?88,.d88b,?88   d8P .d888b,
88P  `88P'd88   88   88P`?88'  ?88d88   88  ?8b,
?8b  d88  ?8(  d88  d88   88b  d8P?8(  d88    `?8b
`?888888  `?88P'?8bd88'   888888P'`?88P'?8b`?888P'
    `?88                  88P'
      88b                d88
      ?8P                ?8P"
               'face '(:foreground "#7c6f64")))
      (insert "\n\n\n")
      (insert (propertize (format-time-string "  %A, %B %-d %Y")
                          'face '(:foreground "#a89984")))
      (insert "\n\n")
      (dolist (cmd '(("C-x C-f"  . "Find file")
                     ("C-x b"    . "Switch buffer")
                     ("C-x C-b"  . "List buffers")
                     ("C-x k"    . "Kill buffer")
                     ("C-s"      . "Search")
                     ("M-x"      . "Run command")
                     ("C-x 1"    . "One window")
                     ("C-x 2"    . "Split horizontal")
                     ("C-x 3"    . "Split vertical")))
        (insert (propertize (format "  %-12s" (car cmd))
                            'face '(:foreground "#fabd2f" :weight bold)))
        (insert (propertize (format "%s\n" (cdr cmd))
                            'face '(:foreground "#665c54"))))
      (insert "\n")
      (insert (propertize "  r  refresh    q  quit\n"
                          'face '(:foreground "#504945"))))
    (read-only-mode 1)
    (goto-char (point-min))
    (use-local-map (make-sparse-keymap))
    (local-set-key (kbd "q") #'bury-buffer)
    (local-set-key (kbd "r") (lambda ()
                               (interactive)
                               (let ((inhibit-read-only t))
                                 (erase-buffer)
                                 (goto-char (point-min))))))
  (setq inhibit-startup-screen t
        initial-scratch-message nil
        initial-buffer-choice (lambda () (get-buffer "*start*"))))
#+END_SRC

*** Org Mode
#+BEGIN_SRC emacs-lisp
(defun my-centered-buffer ()
  "Enable a centered notebook-like layout for writing buffers."
  (setq-local visual-fill-column-width 120
              visual-fill-column-center-text t)
  (visual-fill-column-mode 1)
  (visual-line-mode 1))

(defun my/set-org-fonts ()
  ;; Monospace faces — explicit font instead of relying on inherit
  (set-face-attribute 'line-number nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'line-number-current-line nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'org-block nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'org-code nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'org-verbatim nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'org-block-begin-line nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'org-block-end-line nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'org-table nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
  (set-face-attribute 'org-formula nil
                      :inherit 'fixed-pitch
                      :font "FiraCode Nerd Font" :height 140)
;; Title
  (set-face-attribute 'org-document-title nil
                      :family "NewComputerModern10"
                      :weight 'bold :slant 'italic :height 1.6)
  ;; Headings
  (set-face-attribute 'org-level-1 nil
                      :family "NewComputerModern10"
                      :weight 'bold :slant 'italic :height 1.4)
  (set-face-attribute 'org-level-2 nil
                      :family "NewComputerModern10"
                      :weight 'bold :slant 'italic :height 1.3)
  (set-face-attribute 'org-level-3 nil
                      :family "NewComputerModern10"
                      :weight 'semi-bold :slant 'italic :height 1.2)
  (set-face-attribute 'org-level-4 nil
                      :family "NewComputerModern10"
                      :weight 'normal :slant 'italic :height 1.1))

*** Voice to Text
#+BEGIN_SRC emacs-lisp
(use-package whisper
  :vc (:url "https://github.com/natrys/whisper.el" :branch "master")
  :bind
  ;; Press once to start recording, press again to stop and transcribe
  ("C-c w" . whisper-run)
  :config
  ;; Language model to use. "base" is a good balance of speed and accuracy.
  ;; Options: tiny, base, small, medium, large-v3, large-v3-turbo
  (setq whisper-model "base")
  ;; Your spoken language (ISO 639-1 code)
  (setq whisper-language "en")
  ;; Insert transcribed text at cursor position (set to nil for a separate buffer)
  (setq whisper-insert-text-at-point t)
  ;; Stop recording automatically after 5 minutes if you forget
  (setq whisper-recording-timeout 300))
